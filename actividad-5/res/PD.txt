      2           REAL PROB1(2)=(8.,7.);
      3           REAL D,DCPU,DDISK;
      4           REAL R, RTOTAL;
      5           INTEGER I,N1;
      6 /STATION/ NAME=CPU;
      7 &         SCHED=PS;
      8           SERVICE=EXP(0.03);

     10 /STATION/ NAME=DISC;
     11           TRANSIT=CPU;

     13 /STATION/ NAME=TERMINAL;
     14           TYPE=INFINITE;
     15           INIT=N1;
     16           SERVICE=EXP(8.);

     18 /CONTROL/ CLASS=ALL QUEUE;

     20           BEGIN
     21             PRINT;
     22             PRINT("NUMERO DE USUARIOS=",N1);
     23             SOLVE;
     24  
     25 &           DEMANDA APARTADO 1
     26             DCPU:=MSERVICE(CPU)*PROB1(1);
     27             DDISK:=MSERVICE(DISC)*PROB1(2);
     28             D:=DDISK+DCPU;
     29             PRINT("DEMANDA CPU=",DCPU);
     30             PRINT("DEMANDA DISCO=",DDISK);
     31             PRINT("DEMANDA TOTAL=",D);
     32             PRINT("DEMANDA POR USUARIO=",D/N1);
     33  
     34 &           TIEMPO DE RESPUESTA DEL SISTEMA :
     35             R := MRESPONSE(CPU)*PROB1(1)+MRESPONSE(DISC)*PROB1(2);
     36             RTOTAL := R + MRESPONSE(TERMINAL);
     37             PRINT("TIEMPO DE RESPUESTA DEL SISTEMA=",R);
     38             PRINT("TIEMPO DE RESPUESTA TOTAL DEL SISTEMA=",RTOTAL);
     39             PRINT("TIEMPO DE RESPUESTA DEL CPU=",MRESPONSE(CPU));
     40             PRINT("TIEMPO DE RESPUESTA DEL DISCO=",MRESPONSE(DISC));
     41             PRINT("TIEMPO DE RESPUESTA DEL TERMINAL (TIEMPO DE REFLEXION
)=",MRESPONSE(TERMINAL));
     42  
     43  
     44  
     45           END;
